#version 410 core

layout (vertices = 3) out; //3: how many vertices generated by TCS

in VS_OUT
{
    vec3 pos;
	vec3 norm;
} tcs_in[]; //must be array

out TCS_OUT
{
    vec3 pos;
	vec3 norm;
} tcs_out[]; //must be array

void main(void)
{
    if (gl_InvocationID == 0)
    {
	    gl_TessLevelInner[0] = 5.0;
		gl_TessLevelOuter[0] = 5.0;
		gl_TessLevelOuter[1] = 5.0;
		gl_TessLevelOuter[2] = 5.0;
	}
	gl_out[gl_InvocationID].gl_Position = gl_in[gl_InvocationID].gl_Position;
	tcs_out[gl_InvocationID].pos = tcs_in[gl_InvocationID].pos; //pass through
	tcs_out[gl_InvocationID].norm = tcs_in[gl_InvocationID].norm; //pass through
}
